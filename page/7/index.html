【模拟】A - 小Q系列故事——屌丝的逆袭

题目链接：…… 解释一下： 偏移数组（可能这样叫不合适）：优越性

for(int i=0; i<dx.size(); ++i){
next_x=x+dx[i];
next_y=y+dy[i];
}

循环一边就可以表示周围的点，而变动需要表示的点只需要改动偏移数组 dx[], dy[].   以下是AC程序

#include
using namespace std;
#define MAXN 29
#define inf 1e9+9
int n, m, k[MAXN][MAXN];
int dx[4]={0, 0, -1, 1};
int dy[4]={-1, 1, 0, 0};
struct{
int value, i, j;
}ans;
bool is_in(int x, int y){
return 1<=x&&x<=n&&1<=y&&y<=m;
}
int value(int x, int y){
int r_value=0;//return_value
for(int i=0; i<4; ++i){
if(is_in(x+dx[i], y+dy[i])){
r_value+=k[x+dx[i]][y+dy[i]];
}
}
return k[x][y]<0?r_value:-r_value;
}
void solve(){
for(int i=1; i<=n; ++i){
for(int j=1; j<=m; ++j){
scanf(“%d”, &k[i][j]);
}
}
ans.value=-inf;
for(int i=1; i<=n; ++i){
for(int j=1; j<=m; ++j){
int t_value=value(i, j);
if(t_value>ans.value){
ans.value=t_value;
ans.i=i;
ans.j=j;
}
}
}
printf(“%d %d %d\n”, ans.i, ans.j, ans.value);
}
int main(){
while(scanf(“%d%d”, &n, &m), !(0==n&&0==m)){
solve();
}
return 0;
}

【贪心】浪漫的V8

最简单类型的贪心，优先选择单价最高的金粉即可。

#include
#include
using namespace std;
#define MAXN 1009
int n, V;
struct node{
int v, w;
double u;
}a[MAXN];
bool cmp(node a, node b){
return a.u>b.u;
}
void solve(){
scanf(“%d”, &n);
for(int i=0; i<n; ++i){
scanf(“%d%d”, &a[i].v, &a[i].w);
a[i].u=(double)a[i].w/(double)a[i].v;
}
scanf(“%d”, &V);
sort(a, a+n, cmp);
double ans=0;
for(int i=0; i<n; ++i){
if(V>=a[i].v){
V-=a[i].v;
ans+=(double)a[i].w;
}
else{
ans+=((double)V*a[i].u);
break;
}
}
printf(“%.4lf\n”, ans);
}
int main(){
int T;
scanf(“%d”, &T);
while(T–){
solve();
}
return 0;
}

【模拟】【前缀和】简单逆序对

求逆序对的方法可以用归并排序，时间复杂度为O(nlogn). 注意到此题数字的范围都是 [0, 9]，所以只需要统计 a[i] 之前的比 a[i] 小的数的个数。用前缀和即可。

#include
using namespace std;
#define MOD 1000000007
int tot[10], n, a;
void solve(){
int ans=0;
scanf(“%d”, &n);
for(int i=0; i<=9; ++i){
tot[i]=0;
}
for(int i=1; i<=n; ++i){
scanf(“%d”, &a);
++tot[a];
for(int j=9; j>a; –j){
ans=(ans+tot[j])%MOD;
}
}
printf(“%d\n”, ans);
}
int main(){
int T;
scanf(“%d”, &T);
while(T–){
solve();
}
return 0;
}

    据说用归并排序求逆序对也可以过这道题。

【二分法】【模拟】Arch0n's interesting game

首先将卡牌按分值按从大到小排序。a[1] 是最大的牌，所以不贡献任何值，a[2] 只小于等于 a[1]，所以 a[2] 贡献一个值，一次类推，a[i] 贡献 (i-1) 个值。也就是生成的值的数列是这样的: a2 a3 a3 a4 a4 a4 a5 a5 a5 a5 … 第k大满足 i*(i-1)/2<=k<i*(i+1)/2 找到这个 i 并输出 a[i] 即是答案。 要找到这个i，有两种方法：

  1. 枚举，从 2 到 n 枚举 i， 第一个满足 i*(i-1)/2<=k 的 a[i] 即是答案， O(n)
  2. 二分， O(nlogn)

二分掌握得不熟练，先附上枚举的AC程序。

#include
#include
using namespace std;
#define MAXN 2509
int a[MAXN];
int n, k;
bool cmp(int a, int b){
return a>b;
}
void solve(){
scanf(“%d%d”, &n, &k);
for(int i=1; i<=n; ++i){
scanf(“%d”, &a[i]);
}
sort(a+1, a+1+n, cmp);
for(int i=2; i<=n; ++i){
if(i*(i-1)>=2*k){
printf(“%d\n”, a[i]);
break;
}
}
}
int main(){
int T;
scanf(“%d”, &T);
while(T–){
solve();
}
return 0;
}

【模拟】World Game

注意：1.只有小写字母。2.字母的使用次数有限制

#include
#include
using namespace std;
#define MAXL 29
int n;
char s[MAXL], curs[MAXL];
int osum[26], sum[26];
bool is_ok(char* s){
int len=strlen(s);
for(int i=0; i<26; ++i){
sum[i]=osum[i];
}
for(int i=0; i<len; ++i){
–sum[s[i]-‘a’];
if(sum[s[i]-‘a’]<0){
return false;
}
}
return true;
}
void solve(){
int len=strlen(s);
for(int i=0; i<26; ++i){
osum[i]=0;
}
for(int i=0; i<len; ++i){
++osum[s[i]-‘a’];
}
scanf(“%d”, &n);
int ans=0;
while(n–){
scanf(“%s”, curs);
len=strlen(curs);
if(len>ans){
if(is_ok(curs)){
ans=len;
}
}
}
printf(“%d\n”, ans);
}
int main(){
while(scanf(“%s”, s)!=EOF){
solve();
}
return 0;
}

乘法逆元►数学

【暴力水题】【？乘法逆元】Lnever and Pikachu

题目链接：http://acm.xidian.edu.cn/problem.php?id=1199 思路：起初以为要用Lucas定理，结果发现p太大了。 答案是A(n,m)/m;然后考虑怎么计算，一开始尝试n*(n-1)*…*(n-m+1)边乘边取模，最后再除m，发现样例2过不了，才想到除法不能边算边模。然后想到乘法逆元，杂七杂八的，然后又发现p不一定是质数，反正就是想解决最后一次除法的问题。 百度到p不是质数的一个解决办法（猛然想起以前听人讲过这种情况，好像是要分解p的质因数然后操作一番），讲解是：https://blog.csdn.net/jasonzhu8/article/details/6260245 然后我想这道题怕不是这么难吧，能不能回避这个问题。 然后考虑规避除法问题。 在样例2中 A(100, 50)/m = 100*99*…*51/50%p = 100/50*99*98*…*51%p 可以发现100是50的倍数，刚好规避了除法操作，所以想到在边乘边模的时候就把m给约掉就好了。 但是又会发现不都是这样运气好其中一个因数刚好是m的倍数，所以考虑在边乘边模的过程中求m与因数的最大公约数，一步一步地约掉m，规避除法操作。 AC程序：

#include
using namespace std;
#define ll long long
ll gcd(ll a, ll b){
if(b==0)return a;
return gcd(b, a%b);
}
int main(){
ll n, m, p, cur, g, om;
while(scanf(“%lld%lld%lld”, &n, &m, &p)!=EOF){
int ans=1;
om=m;
for(ll i=n; i>n-om; –i){
cur=i;
g=gcd(m, i);
if(g>1){
cur/=g;
m/=g;
}
ans=(ans*cur)%p;
}
printf(“%lld\n”, ans);
}
return 0;
}

  后话：从实际意义上来说，A(n,m)一定是m的倍数，但是怎么证明呢？   后话2：可以当成乘法逆元题目来做吗？ 挖个坑。

【期望】1phan And His Dice

题目链接：http://acm.xidian.edu.cn/problem.php?id=1285 题目大意： ……   看题解做出来的

#include
using namespace std;
#define maxn 509
double E[maxn];
int main(){
int n;
while(scanf(“%d”, &n)!=EOF){
double ans=0;
for(int i=1; i<=n; ++i){
E[i]=0;
for(int j=1; j<=i; ++j){
E[i]+=(E[i%j]/(double)n);
}
E[i]+=1.0;
E[i]=E[i]*(double)n/(double)i;
ans+=E[i];
//printf(“%d %lf\n”, i, E[i]);
}
printf(“%.2lf\n”, 1+ans/(double)n);
}
return 0;
}
