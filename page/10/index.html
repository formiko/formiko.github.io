Trip For Meal

题目链接：http://codeforces.com/contest/876/problem/A 题目大意：A, B, C 三个点构成，A到B的距离为a，A到C的距离为B，B到C的距离为c，从A出发，总共在n个点上停留，求最小路程ANS

#include<stdio.h>
int min(int a, int b)
{
if(a<b)return a;
return b;
}
int main()
{
int n, a, b, c, ans;
scanf(“%d%d%d%d”, &n, &a, &b, &c);
if(n>2)
{
if(c<a&&c<b)
{
ans=min(a, b)+c*(n-2);
}
else
{
ans=min(a, b)*(n-1);
}
}
else
{
ans=min(a, b)*(n-1);
}
printf(“%d”, ans);
return 0;
}

最小生成树模板题

题目链接（洛谷）https://www.luogu.org/problemnew/show/P3366 裸题，kruscal

#include
#include
using namespace std;
#define maxn 5009
#define maxm 200009
int n, m, father[maxn];
struct node{
int x, y, z;
}edge[maxm];
int getfather(int a){
return father[a]==a?a:father[a]=getfather(father[a]);
}
bool cmp(node a, node b){
return a.z<b.z;
}
void init(){
scanf(“%d%d”, &n, &m);
int x, y, z;
for(int i=0; i<m; ++i){
scanf(“%d%d%d”, &x, &y, &z);
edge[i].x=x;
edge[i].y=y;
edge[i].z=z;;
}
sort(edge, edge+m, cmp);
for(int i=0; i<maxn; ++i){
father[i]=0;
}
}
int kru(){
for(int i=0; i<n; ++i){
father[i]=i;
}
int cnt=1, k=0, ans=0;
while(cnt<n&&k<m){
int x=getfather(edge[k].x), y=getfather(edge[k].y);
if(x!=y){
father[x]=y;
ans+=edge[k].z;
++cnt;
}
++k;
}
if(cnt<n){
return -1;
}
return ans;
}
void solve(){
int ans=kru();
if(-1==ans){
puts(“orz”);
}
else{
printf(“%d\n”, ans);
}
}
int main(){
init();
solve();
return 0;
}

题目链接 【USACO3.1.1】Agri-Net最短网络 prim 裸题 O(n^2)的写法

#include
using namespace std;
#define maxn 109
#define inf 1000000000
int g[maxn][maxn], n, dis[maxn];
int prim(int x){
for(int i=0; i<n; ++i){
dis[i]=g[i][x];
}
int ans=0, mmin, k;
for(int i=1; i<n; ++i){
mmin=inf;
for(int j=0; j<n; ++j){
if(dis[j]&&dis[j]<mmin){
mmin=dis[j];
k=j;
}
}
ans+=mmin;
dis[k]=0;
for(int j=0; j<n; ++j){
if(g[j][k]<dis[j]){
dis[j]=g[j][k];
}
}
}
return ans;
}
void init(){
scanf(“%d”, &n);
for(int i =0; i<n; ++i){
for(int j=0; j<n; ++j){
scanf(“%d”, &g[i][j]);
}
}
}
void solve(){
printf(“%d\n”, prim(0));
}
int main(){
init();
solve();
return 0;
}

ios_base :: sync_with_stdio(false);和cin.tie(0)的用法

在 codeforces 上看到有人这样写，据说这叫关闭同步流，效果就是使得 cin, cout的速度接近于 scanf, printf.

#include
#include
using namespace std;
int main(){
ios_base :: sync_with_stdio(false);
cin.tie(0);
int a;
cin>>a;
cout<<a;
return 0;
}

N！的末尾的零的个数

 

1019: 自然数的秘密

时间限制: 2 Sec  内存限制: 128 MB

题目描述

题意： 已知：N！=N*(N-1)*…*2*1 找到最小自然数 N, 使N!末尾有连续 M个零. 例如, 5! 的结尾包含1个零.

输入

第一行输入一个整数T,表示有T组测试数据。 对于每组测试数据，输入一个整数M，表示包含M个零。（0<=M<=10^8）

输出

每组数据，输出一行满足条件的最小自然数N。 如果无解，输出“No solution”。（不含引号）

样例输入

2
1
2

样例输出

5
10

题目来源：http://acm.xidian.edu.cn/problem.php?id=1019   正解：log5(N)的复杂度可得到N!末尾零的个数，二分答案。 最大答案=MA=400000015 O（T*log2(MA)*log5(MA)）

#include<stdio.h>
#define MAXANS 500000000
int z(int n)
{
int t=0;
while(n)
{
t+=n/5;
n/=5;
}
return t;
}
void work(int m)
{
int l=0, r=MAXANS, mid, cur;
while(l<r)
{
mid=(l+r)/2;
cur=z(mid);
if(cur>=m) r=mid;
else if(cur<m) l=mid+1;
}
if(z(l)==m) printf(“%d\n”, l);
else printf(“No solution\n”);
}
int main()
{
int t, m, n, i;
scanf(“%d”, &t);
while(t–)
{
scanf(“%d”, &m);
work(m);
}
return 0;
}

  求N！末尾零的个数的方法：

  1. O(N), 算出N!的值，然后%10得到末尾0的个数，但是N!增长极快。
  2. O(log5(N)), 考虑N!的末尾零的产生原因是5的公倍数，所以只需要计算出1~N有多少个因数5就可以了 tot=N/5+N/5^2+…+N/5^k

int ZeroNum(int n)
{
int tot=0;
while(n)
{
tot+=n/5;
n/=5;
}
return tot;
}

  插曲：一开始这道题我的做法是枚举N=0~5亿，然后打表前缀和，把5亿分成100个相等的区间。然而还是妥妥的超时了，程序还特别冗杂易错。
