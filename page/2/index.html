// formiko learn f-zyj’s tmp
#include
#include
#include
using namespace std;
const int MAXN = 1e4+9;
const int MAXLEN = 53;
const int ALPHA = 26;
struct Trie{
int next[MAXN*MAXLEN][ALPHA], fail[MAXN*MAXLEN], end[MAXLEN*MAXN];
int root, L;
int newnode(){
for(int i=0; i<26; ++i){
next[L][i]=-1;
}
end[L++]=0;
return L-1;
}
void init(){
L=0;
root=newnode();
}
void insert(char buf[]){
int len=(int)strlen(buf);
int now=root;
for(int i=0; i<len; ++i){
if(-1==next[now][buf[i]-‘a’]){
next[now][buf[i]-‘a’]=newnode();
}
now=next[now][buf[i]-‘a’];
}
++end[now];
}
void build(){
queueQ;
fail[root]=root;
for(int i=0; i<26; ++i){
if(-1==next[root][i]){
next[root][i]=root;
}
else{
fail[next[root][i]]=root;
Q.push(next[root][i]);
}
}
while(!Q.empty()){
int now=Q.front();
Q.pop();
for(int i=0; i<26; ++i){
if(-1==next[now][i]){
next[now][i]=next[fail[now]][i];
}
else{
fail[next[now][i]]=next[fail[now]][i];
Q.push(next[now][i]);
}
}
}
}
int query(char buf[]){
int len=(int)strlen(buf);
int now=root;
int res=0;
for(int i=0; i<len; ++i){
now=next[now][buf[i]-‘a’];
int temp=now;
while(temp!=root){
res+=end[temp];
end[temp]=0;
temp=fail[temp];
}
}
return res;
}
};
char buf[MAXN*MAXLEN*2];
Trie ac;
void solve(){
int n;
scanf(“%d”, &n);
ac.init();
for(int i=0; i<n; ++i){
scanf(“%s”, buf);
ac.insert(buf);
}
ac.build();
scanf(“%s”, buf);
printf(“%d\n”, ac.query(buf));
}
int main(){
int T;
scanf(“%d”, &T);
while(T–){
solve();
}
return 0;
}

void phi_table(int n, int phi[]){
for(int i=2; i<=n; ++i){
phi[i]=0;
}
phi[1]=1;
for(int i=2; i<=n; ++i){
if(!phi[i]){
for(int j=i; j<=n; j+=i){
if(!phi[j]){
phi[j]=j;
}
phi[j]=phi[j]/i*(i-1);
}
}
}
}

void better_phi_table(int n, int phi[]){
for(int i=1; i<=n; ++i){
phi[i]=i;
}
for(int i=2; i<=n; i+=2){
phi[i]/=2;
}
for(int i=3; i<=n; i+=2){
if(phi[i]==i){
for(int j=i; j<=n; j+=i){
phi[j]=phi[j]/i*(i-1);
}
}
}
}

第二种写法更优。 maxn=1e7 第一种写法 0.6s， 第二种写法 0.5s maxn=1e8 第一种写法 6s，第二种写法5s

void Eratosthenes(int n, bool vis[]){
for(int i=0; i<n; ++i){
vis[i]=false;
}
for(int i=2; i<=n; ++i){
if(!vis[i]){
for(int j=i*i; j<=n; j+=i){
vis[j]=true;
}
}
}
}

/*
phi(x)=n(1-1/p1)(1-1/p2)…(1-1/pk)
=n(/p1*(p1-1))(/p2*(p2-1))…(/pk*(pk-1))
pi是x的第i个质因数

int euler_phi(int x){
int ret=x;
for(int i=2; i*i<=x; ++i){//写成 i*i<=x，x在不断变小，相比 i<=sqrt_x, 可能是一个很强的优化（本机效果是提速2/3)

ret=ret/x*(x-1);
}
return ret;
}

sqrt(n), sqrt(n+0.5), sqrt(n)+0.5

分解x的质因数或求单个欧拉函数的的时候会枚举因子，据说写sqrt(n)会因为精度问题出现 .9999999变成实际值-1的值导致出错。 测试了一下发现好像在int范围内用于分解质因数并没有什么区别。 n为int类型非负整数 sqrt(n), sqrt(n+0.5), sqrt(n)+0.5 的大小关系 double类型下: sqrt(n)<sqrt(n+0.5)<sqrt(n)+0.5 强制转换为int类型： sqrt(n)=sqrt(n+0.5)<=sqrt(n)+0.5 运行以下程序检验：

#include
#include
using namespace std;
int main(){
int i;
for(int i=0; i>=0; ++i){
if((int)sqrt(i)!=(int)sqrt(i+0.5)(int)sqrt(i)>(int)(sqrt(i)+0.5)){
printf(“%d\n”, i);
}
}
putchar(‘*‘);
return 0;
}

结果如下 说明在int范围内成立。 分解质因数可以写成

int sqr_x=sqrt(x);
for(int i=2; i<=sqr_x; ++i){
if(0==x%i){
d[tot++]=i;
}
while(0==x%i){
x/=i;
}
}

当然其实我还是觉得不写sqrt，直接 i*i<=x 更优，因为x在不断变小，这样可能可以提前终止循环。

int sqr_x=sqrt(x);
for(int i=2; i*i<=x; ++i){
if(0==x%i){
d[tot++]=i;
}
while(0==x%i){
x/=i;
}
}

  可能并没有什么营养的 Point 反正优先用 i*i<=x，不行就用 sqrt(x)，暂时没发现用后两者的必要。 至于 i*i 的优越性 单个计算欧拉函数打表[0, 1e6]， i*i 的写法大概是 sqrt 写法的 1/3（sqrt写法要3.8s左右，i*i写法要1.1s左右），求[0, 1e7]，两种写法都跑不出来。

几个排列组合计数的变式问题

原问题： 可重复选择的组合。有n个不同元素，每个元素可以选多次，一共选k个元素，有多少种方法。 刘汝佳; 陈锋. 算法竞赛入门经典——训练指南 (算法艺术与信息学竞赛) (Kindle位置1683). 清华大学出版社. Kindle 版本. 解：ANS = C(n+k-1,  n-1)。可以用方程的解的思路来做。 变式：

  1. 有n个不同元素，每个元素可以选多次，一共选k个元素排成一排，有多少种排法。
  2. 有n个元素，每个元素有ni个，每个元素可以选[0,ni]个，一共选k个元素，有多少种方法。
  3. 有n个元素，每个元素有ni个，每个元素可以选[0,ni]个，一共选k个元素排成一排，有多少种排法。
