【模拟】World Game

注意：1.只有小写字母。2.字母的使用次数有限制

#include
#include
using namespace std;
#define MAXL 29
int n;
char s[MAXL], curs[MAXL];
int osum[26], sum[26];
bool is_ok(char* s){
int len=strlen(s);
for(int i=0; i<26; ++i){
sum[i]=osum[i];
}
for(int i=0; i<len; ++i){
–sum[s[i]-‘a’];
if(sum[s[i]-‘a’]<0){
return false;
}
}
return true;
}
void solve(){
int len=strlen(s);
for(int i=0; i<26; ++i){
osum[i]=0;
}
for(int i=0; i<len; ++i){
++osum[s[i]-‘a’];
}
scanf(“%d”, &n);
int ans=0;
while(n–){
scanf(“%s”, curs);
len=strlen(curs);
if(len>ans){
if(is_ok(curs)){
ans=len;
}
}
}
printf(“%d\n”, ans);
}
int main(){
while(scanf(“%s”, s)!=EOF){
solve();
}
return 0;
}
