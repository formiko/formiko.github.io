动态规划

【递推】【动态规划】LCIS

假的最长上升公共子序列。为什么是假的呢？因为这个子序列的值是连续的 (lll￢ω￢)） 一开始没注意到必须连续这个点，一度怀疑狗生。 所以此题只需要分开找每个序列的最长上升连续子序列，然后比对一下就行了。 本来这样就OK了，由于我在一开始想这道题一直在列两序列相关的状态方程，影响了之后的思路，思维僵化，成功的把此题做得相当复杂，记录了值 a[i] 最后出现的位置pa，反正多此一举，搞得常数超大。 场面一度十分尴尬。 然后加了初始化优化：不清空所有pa, pb，而是清空 min_v 到 max_v 之间的 pa[] 和 pb[]。 附上优化后的超丑程序。

#include
using namespace std;
#define MAXN 1000009
int a[MAXN], b[MAXN], dpa[MAXN], dpb[MAXN], pa[MAXN], pb[MAXN];
int max(int a, int b){
return a>b?a:b;
}
int min(int a, int b){
return a<b?a:b;
}
void solve(){
int n, m, min_v=MAXN, max_v=0;
scanf(“%d%d”, &n, &m);
for(int i=1; i<=n; ++i){
scanf(“%d”, &a[i]);
min_v=min(min_v, a[i]);
max_v=max(max_v, a[i]);
}
for(int i=1; i<=m; ++i){
scanf(“%d”, &b[i]);
min_v=min(min_v, b[i]);
max_v=max(max_v, b[i]);
}
for(int i=min_v-1; i<=max_v; ++i){
pa[i]=0;
pb[i]=0;
}
for(int i=1; i<=n; ++i){
dpa[i]=dpa[pa[a[i]-1]]+1;
pa[a[i]]=i;
}
for(int i=1; i<=m; ++i){
dpb[i]=dpb[pb[b[i]-1]]+1;
pb[b[i]]=i;
}
int ans=0;
for(int i=1; i<=n; ++i){
ans=max(ans, min(dpa[pa[a[i]]], dpb[pb[a[i]]]));
}
printf(“%d\n”, ans);
}
int main(){
int T;
scanf(“%d”, &T);
while(T–){
solve();
}
return 0;
}

  不优化多交几次的话（然而真的太靠运气了）  

相当惨，后来看了 https://blog.csdn.net/qq_40772738/article/details/80572962 的程序才知道自己实现得太复杂了。 正确的姿势如下。

#include
#include
using namespace std;
#define MAXN 100009
int n, m, a[MAXN], b[MAXN];
int min(int a, int b){
return a<b?a:b;
}
int max(int a, int b){
return a>b?a:b;
}
void solve(){
scanf(“%d%d”, &n, &m);
int t;
memset(a, 0, sizeof(a));
memset(b, 0, sizeof(b));
for(int i=1; i<=n; ++i){
scanf(“%d”, &t);
a[t]=a[t-1]+1;
}
for(int i=1; i<=m; ++i){
scanf(“%d”, &t);
b[t]=b[t-1]+1;
}
int ans=0;
for(int i=1; i<MAXN; ++i){
ans=max(ans, min(a[i], b[i]));
}
printf(“%d\n”, ans);
}
int main(){
int T;
scanf(“%d”, &T);
while(T–){
solve();
}
return 0;
}

累觉不爱。 原来求单个序列的最长上升连续子序列只需要一句 a[t] = a[t-1]+1     所以这道题和动态规划有什么关系，纯递推呀
