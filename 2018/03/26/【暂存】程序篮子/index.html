#include<stdio.h>
#include<string.h>
#define maxl 1009
#define ll long long
char a[maxl];//用字符串存储m
ll m[maxl], n, ans[maxl];
void solve1();//解决乘法情况
void solve2(); //解决除法情况
void init(); //初始化
int main(){
int sel;
while(1){
puts(“\n请选择: (0/1/2):”);
printf(“\t(1)长整数乘普通整数\n”);
printf(“\t(2)长整数除普通整数\n”);
printf(“\t(0)退出\n”);
scanf(“%d”, &sel);
if(1==sel){
solve1();
}
else if(2==sel){
solve2();
}
else{
break;
}
}

}

void init(){
int i=0;
for(i=0; i<maxl; ++i){
ans[i]=0;
}
}
void solve1(){
int len, i;
init();
puts(“将进行乘法运算，请输入长整数 m :”);
scanf(“%s”, a);
puts(“请输入普通整数 n (n在int类型的表示范围之内):”);
scanf(“%I64d”, &n);
if(n<0){
putchar(‘-‘);
n=-n;
}
len=strlen(a);
for(i=0; i<len; ++i){
m[i]=(ll)(a[i]-‘0’);
}
for(i=len-1; i; –i){
ans[i]+=m[i]*n;
ans[i-1]+=ans[i]/10;
ans[i]%=10;
}
ans[0]+=m[0]*n;
for(i=0; i<len; ++i){
printf(“%d”, ans[i]);
}
}
void solve2(){
ll lm;
double m, ans;
init();
puts(“将进行除法运算，请输入长整数 m :”);
scanf(“%I64d”, &lm);
puts(“请输入普通整数 n (n在int类型的表示范围之内):”);
scanf(“%I64d”, &n);
if(n<0){
putchar(‘-‘);
n=-n;
}
printf(“%lf”, (double)(lm/n)+(double)(lm%n)/(double)n);
}

void kmp(){
int n, m, i, j;
a[0]=’*‘;
b[0]=’*‘;
n=strlen(a)-1, m=strlen(b)-1;
p[1]=0;
j=0;
for(i=2; i<=m; ++i){
while(j>0&&b[j+1]!=b[i])j=p[j];
if(b[j+1]==b[i])++j;
p[i]=j;
}
j=0;
int lastp=1;
for(i=1; i<=n; ++i){
while(j>0&&b[j+1]!=a[i])j=p[j];
if(b[j+1]==a[i])++j;
if(j==m){
//printf(“***%d\n”, i-m+1);
for(int k=lastp; k<=i-m; ++k){
putchar(a[k]);
}
printf(“%s”, c+1);
lastp=i+1;
j=p[j];
}
}
for(int k=lastp; k<=n; ++k){
putchar(a[k]);
}
}

#include<stdio.h>
#include<string.h>
#define maxn 10009
struct node{
char name[30], email[30], path[100], sex[20], pho[20];
int used;
}a[maxn];
int tail;
void find(int key, char* s){
int i=0;
char *c;
for(i=0; i<tail; ++i){
if(a[i].used){
if(1==key){
c=a[i].name;
}
else if(2==key){
c=a[i].pho;
}
else if(3==key){
c=a[i].email;
}
else if(4==key){
c=a[i].path;
}
if(!strcmp(c, s)){
puts(“*************************“);
printf(“%s\n%s\n%s\n%s\n%s\n”, a[i].name, a[i].sex, a[i].pho, a[i].email, a[i].path);
puts(“*************************“);
}
}
}
}
void del(int key, char* s){
int i=0;
char *c;
for(i=0; i<tail; ++i){
if(a[i].used){
if(1==key){
c=a[i].name;
}
else if(2==key){
c=a[i].pho;
}
else if(3==key){
c=a[i].email;
}
else if(4==key){
c=a[i].path;
}
if(!strcmp(c, s)){
a[i].used=false;
}
}
}
}

#include<stdio.h>
#include<time.h>
#include<stdlib.h>
#include<math.h>
int main(){
int n, op, n1, n2, tot=0;
double ans, realans;
srand((unsigned) time(NULL));
puts(“小学生数学测试软件”);
puts(“请输入答题的个数:”);
scanf(“%d”, &n);
while(n–){
n1=rand()%100+1;
n2=rand()%100+2;
op=rand()%4+1;
printf(“%d “, n1);
if(1==op)putchar(‘+’), realans=(double)n1+(double)n2;
else if(2==op)putchar(‘-‘), realans=(double)n1-(double)n2;
else if(3==op)putchar(‘*‘), realans=(double)n1*(double)n2;
else if(4==op)putchar(‘/‘), realans=(double)n1/(double)n2;
printf(“ %d=?\n”, n2);
puts(“请输入答案”);
scanf(“%llf”, &ans);
if(abs(abs(ans)-abs(realans))<0.001){
tot++;
puts(“答对了”);
}
else{
puts(“答错了”);
}
}
printf(“总共答对了 %d 道题目\n”, tot);
return 0;
}
